docker run -d -e PUID=${PUID} \
              -e PGID=${PGID} \
              -e POSTGRES_DB=nextcloud \
              -e POSTGRES_USER=nextcloud \
              -e POSTGRES_PASSWORD= \
              -v ${USERDIR}/postgresdb:/var/lib/postgresql/data \
              -v "/etc/localtime:/etc/localtime:ro" \
              --restart=on-failure:10 \
              --network=webproxy \
              --name postgres \
              postgres:alpine
